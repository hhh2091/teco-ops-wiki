## 16.4.2.2 矩阵乘法单元（续）

### 矩阵乘法单元内核实现片段（接续）

以下为「矩阵乘法单元」优化策略内核实现的尾部代码片段，接续上一部分的内核实现（SPM 空间分配、句柄初始化、分块遍历）。

```cpp
    half *LocalCompB = (half *)malloc(LenB * 2);
    // 是存储 B 矩阵的转置
    half *LocalTB = LocalCompB + LocalbK * LocalbN;
    half *LocalC = (half *)malloc(LenC);
    // 存储中间结果
    float *LocalCompC = (float *)malloc(LocalbM * LocalbN * sizeof(float) * 2);
    float *tempC = LocalCompC + LocalbM * LocalbN;
    half *pCA = LocalCompA;
    half *pCB = LocalCompB;

    int im, in, ik;
    int idM, idN, idK;

    SIMDType vtb[SIMDSIZE];

    // 开始性能采样
    perf_start();

    // 初始化矩阵乘法句柄
    MatmulHandle handleMMA;
    matmul_init(handleMMA, MatmulHalfToFloat);

    // 遍历矩阵的分块
    for (int idx = 0; idx < nM * nN; ++idx) {
        // 计算当前分块的行索引和列索引
        idM = idx / nN;
        idN = idx % nN;
        for (idK = 0; idK < nK; ++idK) {

                simd_load(vtb[10], (float *)(pCB + (ik + 5) * LocalbN));
                simd_load(vtb[11], (float *)(pCB + (ik + 5) * LocalbN + 32));
                simd_load(vtb[12], (float *)(pCB + (ik + 6) * LocalbN));

                    matmul_set_output_row_offset(handleMMA, MatmulEnableOutputRowOffset, LocalbM);
                    matmul_load_weight(handleMMA,

                                       MatmulK32, MatmulN32);
                    matmul_compute(handleMMA, pCA + ik, LocalbM, MatmulK32, LocalbK / REDUITEM - 1);
                }
            } else {
                for (ik = 0; ik + 64 - 1 < LocalbK; ik += 32) {
                    matmul_set_flushing_output(handleMMA, false);
                    matmul_set_output_row_offset(handleMMA, MatmulEnableOutputRowOffset, 0);
                    matmul_load_weight(handleMMA, LocalTB + ik * LocalbN / 2, MatmulK32, MatmulN32);
                    matmul_compute(handleMMA, pCA + ik, LocalbM, MatmulK32, LocalbK / REDUITEM - 1);

                    matmul_set_flushing_output(handleMMA, false);
                    matmul_set_output_row_offset(handleMMA, MatmulEnableOutputRowOffset, LocalbM);
                    matmul_load_weight(handleMMA,
                                       LocalTB + LocalbN * LocalbK / 2 + ik * LocalbN / 2,
                                       MatmulK32, MatmulN32);
                    matmul_compute(handleMMA, pCA + ik, LocalbM, MatmulK32, LocalbK / REDUITEM - 1);
                }
                matmul_set_flushing_output(handleMMA, false);
                matmul_set_output_row_offset(handleMMA, MatmulEnableOutputRowOffset, 0);
                matmul_load_weight(handleMMA, LocalTB + ik * LocalbN / 2, MatmulK32, MatmulN32);
                matmul_compute(handleMMA, pCA + ik, LocalbM, MatmulK32, LocalbK / REDUITEM - 1);

                matmul_set_flushing_output(handleMMA, true);
                matmul_set_output_row_offset(handleMMA, MatmulEnableOutputRowOffset, LocalbM);
                matmul_load_weight(handleMMA, LocalTB + LocalbN * LocalbK / 2 + ik * LocalbN / 2,
                                   MatmulK32, MatmulN32);
                matmul_compute(handleMMA, pCA + ik, LocalbM, MatmulK32, LocalbK / REDUITEM - 1);
                // 存储矩阵乘法结果
                matmul_store(handleMMA, LocalCompC, LocalbM * 2, MatmulN32);
            }
        }    // loop idk
        // 等待矩阵乘法操作完成
        matmul_wait(handleMMA);

        // 对结果 C 进行置换操作
        for (im = 0; im < LocalbM; im++) {
            simd_load(vtb[0], LocalCompC + im * 32);
            simd_load(vtb[1], LocalCompC + im * 32 + 16);
            simd_load(vtb[2], LocalCompC + (im + LocalbM) * 32);
            simd_load(vtb[3], LocalCompC + (im + LocalbM) * 32 + 16);
            simd_store(vtb[0], tempC + im * 64);
            simd_store(vtb[1], tempC + im * 64 + 16);
            simd_store(vtb[2], tempC + im * 64 + 32);
            simd_store(vtb[3], tempC + im * 64 + 48);
        }

        // 在存储结果之前，将计算结果进行类型转换
        batch_S2H(tempC, (half *)LocalC, 4096);

        // 将结果复制回矩阵 C 的对应位置
        memcpy_stride(pC + idM * bM * ldc + idN * bN, LocalC, LenC, BsizeC, {(size_t)StrideC, 0});
    }   // Loop idx

    // 性能采样
    perf_stop();
    // 打印出性能采样结果
    perf_print();

    // 释放分配的内存
    free(LocalCompA);
    free(LocalCompB);
    free(LocalC);
    free(LocalCompC);
}
```

### 16.4.2.2.2 完整程序

使用矩阵乘法单元优化策略的完整程序（`gemm_matmul.scpp`）：

```cpp
#include <sdaa_matmul.h>
#include <sdaa_perf.h>
#include <type_traits>

using namespace sdaa;

#define REDUITEM 32
#define SIMDSIZE 32
typedef floatv16 SIMDType;

// GEMMArgs包含矩阵乘法所需的参数
typedef struct GEMMArgs {
    int m;    // 矩阵 A 与矩阵 C 的行数
    int n;    // 矩阵 B 与矩阵 C 的列数
    int k;    // 矩阵 A 的列数与矩阵 B 的行数
    int lda;   // 矩阵 A 的列间距
    int ldb;   // 矩阵 B 的列间距
    int ldc;   // 矩阵 C 的列间距
    int bM;    // 分块矩阵的行数
    int bN;    // 分块矩阵的列数
    int bK;    // 分块矩阵的中间维度
    const void *A;    // 矩阵 A 的指针
    const void *B;    // 矩阵 B 的指针
    void *C;   // 矩阵 C 的指针
} GEMMArgs;

// 设备函数，将浮点数数组转换为半精度浮点数数组，并根据对齐情况进行处理
__device__ void align_batch_function(float* x, half* y, int len)
{
    // 定义向量变量
    floatv16 vx0, vx1, vx2, vx3, vx4, vx5, vx6, vx7;
    halfv16 vy0, vy1, vy2, vy3, vy4, vy5, vy6, vy7;
    int i = 0;

    // 外层循环，以 128 个元素为一组处理数据
    // 每次循环处理完一组后，指针移动 128 个元素位置继续处理下一组
    for (i = 0; i <= len - 128; i += 128) {
        // 使用 simd_load 函数加载数据
        simd_load(vx0, x);
        simd_load(vx1, x + 16);
        simd_load(vx2, x + 32);
        simd_load(vx3, x + 48);
        simd_load(vx4, x + 64);
        simd_load(vx5, x + 80);
        simd_load(vx6, x + 96);
        simd_load(vx7, x + 112);

        // 使用 simd_cvt_f2h 函数将加载的浮点型向量数据转换为半精度浮点型
        vy0 = simd_cvt_f2h(vx0);
        vy1 = simd_cvt_f2h(vx1);
        vy2 = simd_cvt_f2h(vx2);
        vy3 = simd_cvt_f2h(vx3);
        vy4 = simd_cvt_f2h(vx4);
        vy5 = simd_cvt_f2h(vx5);
        vy6 = simd_cvt_f2h(vx6);
        vy7 = simd_cvt_f2h(vx7);

        // 使用 simd_store 函数将转换后的半精度浮点型向量数据存储到内存中
        simd_store(vy0, y);
        simd_store(vy1, y + 16);
        simd_store(vy2, y + 32);
        simd_store(vy3, y + 48);
        simd_store(vy4, y + 64);
        simd_store(vy5, y + 80);
        simd_store(vy6, y + 96);
        simd_store(vy7, y + 112);

        // 更新指针
        x += 128;
        y += 128;
    }

    // 中层循环，处理长度小于 128 但大于或等于 64 的剩余数据
    // 每次以 64 个元素为一组进行处理
    // 每次循环处理完一组后，指针移动 64 个元素位置继续处理下一组
    for (; i <= len - 64; i += 64) {
        simd_load(vx0, x);
        simd_load(vx1, x + 16);
        simd_load(vx2, x + 32);
        simd_load(vx3, x + 48);

        vy0 = simd_cvt_f2h(vx0);
        vy1 = simd_cvt_f2h(vx1);
        vy2 = simd_cvt_f2h(vx2);
        vy3 = simd_cvt_f2h(vx3);

        simd_store(vy0, y);
        simd_store(vy1, y + 16);
        simd_store(vy2, y + 32);
        simd_store(vy3, y + 48);

        x += 64;
        y += 64;
    }

    // 内层循环，处理长度小于 64 但大于或等于 16 的剩余数据
    // 每次以 16 个元素为一组进行处理
    for (; i <= len - 16; i += 16) {
        simd_load(vx0, x);
        vy0 = simd_cvt_f2h(vx0);
        simd_store(vy0, y);

        x += 16;
        y += 16;
    }

    // 最内层循环，处理长度小于 16 的剩余数据
    // 逐元素进行赋值操作
    for (; i < len; i++) {
        *y = *x;
        x++;
        y++;
    }
}

// 根据长度和对齐情况调用 align_batch_function 进行浮点数到半精度浮点数的转换
__device__ void batch_S2H(float* x, half* y, int len)
{
    if (len < 32) {
        for (int i = 0; i < len; i++) {
            y[i] = x[i];
        }
        return;
    }
    const int align_n1 = sizeof(floatv16);
    const int align_n2 = sizeof(halfv16);
    if ((((size_t)x & (align_n1 - 1)) == 0) && ((((size_t)y) & (align_n2 - 1)) == 0)) {
        align_batch_function((float*)x, y, len);
        return;
    } else {
        int padding_x = ((size_t)(x) & (align_n1 - 1)) / sizeof(float);
        int padding_y = ((size_t)(y) & (align_n2 - 1)) / sizeof(half);
        if (padding_x == padding_y) {
            for (int i = 0; i < 16 - padding_x; i++) {
                y[i] = x[i];
            }
            align_batch_function(x + 16 - padding_x, y + 16 - padding_y, len - 16 + padding_x);
            return;
        } else {
            align_batch_function((float*)x, y, len);
            return;
        }
    }
}

__global__ void tecoKernelGemmMatmulImpl(GEMMArgs pGemm)
{
    // 计算线程行索引和列索引
    const int rid = threadIdx / 8;
    const int cid = threadIdx % 8;

    // 从 GEMMArgs 结构体中获取矩阵的维度信息
    const int M = pGemm.m;
    const int N = pGemm.n;
    const int K = pGemm.k;

    // 从 GEMMArgs 结构体中获取矩阵 A、B、C 的存储步长
    const int lda = pGemm.lda;
    const int ldb = pGemm.ldb;
    const int ldc = pGemm.ldc;

    // 获取矩阵 A、B、C 的指针
    const half *A = (const half *)pGemm.A;
    const half *B = (const half *)pGemm.B;
    half *C = (half *)pGemm.C;

    // 外部分块的大小
    const int bM = pGemm.bM;
    const int bN = pGemm.bN;
    const int bK = pGemm.bK;

    // 外部分块的数量
    int nM = M / bM;
    int nN = N / bN;
    int nK = K / bK;

    // 内部块的大小
    int LocalbM = bM / 8;
    int LocalbN = bN / 4;
    int LocalbK = bK;

    // 内部块的位置
    const half *pA = A + cid * LocalbM * lda;
    const half *pB = B + rid * LocalbN;
    half *pC = C + cid * LocalbM * ldc + rid * LocalbN;

    // 指向当前操作元素的指针
    const half *pCurrA;
    const half *pCurrB;
    half *pCurrC;

    // 计算存储数据所需的长度和步长，考虑元素大小和存储布局
    const int LenA = LocalbM * LocalbK * sizeof(half);
    const int BsizeA = LocalbK * sizeof(half);
    const int StrideA = (lda - LocalbK) * sizeof(half);

    const int LenB = LocalbK * LocalbN * sizeof(half);
    const int BsizeB = LocalbN * sizeof(half);
    const int StrideB = (ldb - LocalbN) * sizeof(half);

    const int LenC = LocalbM * LocalbN * sizeof(half);
    const int BsizeC = LocalbN * sizeof(half);
    const int StrideC = (ldc - LocalbN) * sizeof(half);

    // 计算所需的SPM存储空间大小
    int spm_size = LenA + 2 * LenB + LenC + LenC / sizeof(half) * sizeof(float) * 2;

    // 分配SPM存储空间
    half *LocalCompA = (half *)malloc(LenA);
    half *LocalCompB = (half *)malloc(LenB * 2);
    // 是存储 B 矩阵的转置
    half *LocalTB = LocalCompB + LocalbK * LocalbN;
    half *LocalC = (half *)malloc(LenC);

    // 存储中间结果
    float *LocalCompC = (float *)malloc(LocalbM * LocalbN * sizeof(float) * 2);
    float *tempC = LocalCompC + LocalbM * LocalbN;
    half *pCA = LocalCompA;
    half *pCB = LocalCompB;

    int im, in, ik;
    int idM, idN, idK;

    SIMDType vtb[SIMDSIZE];

    // 开始性能采样
    perf_start();

    // 初始化矩阵乘法句柄
    MatmulHandle handleMMA;
    matmul_init(handleMMA, MatmulHalfToFloat);

    // 遍历矩阵的分块
    for (int idx = 0; idx < nM * nN; ++idx) {
        // 计算当前分块的行索引和列索引
        idM = idx / nN;
        idN = idx % nN;
        for (idK = 0; idK < nK; ++idK) {
            // 获取矩阵 A 和 B 中当前分块的位置
            pCurrA = pA + idM * bM * lda + idK * bK;   // A[idM][idK]
            pCurrB = pB + idK * bK * ldb + idN * bN;   // B[idK][idN]

            // 等待加载输入到矩阵乘法句柄
            matmul_wait_loading_input(handleMMA);
            // 将数据复制到SPM存储空间
            memcpy_stride(pCA, pCurrA, LenA, BsizeA, {0, (size_t)StrideA});

            // 等待加载权重到矩阵乘法句柄
            matmul_wait_loading_weight(handleMMA);
            // 将数据复制到SPM存储空间
            memcpy_stride(pCB, pCurrB, LenB, BsizeB, {0, (size_t)StrideB});

            // 使用 SIMD 加载和存储数据，对 B 矩阵进行转置存储
            for (ik = 0; ik < LocalbK; ik += 8) {
                simd_load(vtb[0], (float *)(pCB + ik * LocalbN));
                simd_load(vtb[1], (float *)(pCB + ik * LocalbN + 32));
                simd_load(vtb[2], (float *)(pCB + (ik + 1) * LocalbN));
                simd_load(vtb[3], (float *)(pCB + (ik + 1) * LocalbN + 32));
                simd_load(vtb[4], (float *)(pCB + (ik + 2) * LocalbN));
                simd_load(vtb[5], (float *)(pCB + (ik + 2) * LocalbN + 32));
                simd_load(vtb[6], (float *)(pCB + (ik + 3) * LocalbN));
                simd_load(vtb[7], (float *)(pCB + (ik + 3) * LocalbN + 32));
                simd_load(vtb[8], (float *)(pCB + (ik + 4) * LocalbN));
                simd_load(vtb[9], (float *)(pCB + (ik + 4) * LocalbN + 32));
                simd_load(vtb[10], (float *)(pCB + (ik + 5) * LocalbN));
                simd_load(vtb[11], (float *)(pCB + (ik + 5) * LocalbN + 32));
                simd_load(vtb[12], (float *)(pCB + (ik + 6) * LocalbN));
                simd_load(vtb[13], (float *)(pCB + (ik + 6) * LocalbN + 32));
                simd_load(vtb[14], (float *)(pCB + (ik + 7) * LocalbN));
                simd_load(vtb[15], (float *)(pCB + (ik + 7) * LocalbN + 32));
                simd_store(vtb[0], (float *)(LocalTB + ik * 32));
                simd_store(vtb[1], (float *)(LocalTB + 32 * LocalbK + ik * 32));
                simd_store(vtb[2], (float *)(LocalTB + (ik + 1) * 32));
                simd_store(vtb[3], (float *)(LocalTB + 32 * LocalbK + (ik + 1) * 32));
                simd_store(vtb[4], (float *)(LocalTB + (ik + 2) * 32));
                simd_store(vtb[5], (float *)(LocalTB + 32 * LocalbK + (ik + 2) * 32));
                simd_store(vtb[6], (float *)(LocalTB + (ik + 3) * 32));
                simd_store(vtb[7], (float *)(LocalTB + 32 * LocalbK + (ik + 3) * 32));
                simd_store(vtb[8], (float *)(LocalTB + (ik + 4) * 32));
                simd_store(vtb[9], (float *)(LocalTB + 32 * LocalbK + (ik + 4) * 32));
                simd_store(vtb[10], (float *)(LocalTB + (ik + 5) * 32));
                simd_store(vtb[11], (float *)(LocalTB + 32 * LocalbK + (ik + 5) * 32));
                simd_store(vtb[12], (float *)(LocalTB + (ik + 6) * 32));
                simd_store(vtb[13], (float *)(LocalTB + 32 * LocalbK + (ik + 6) * 32));
                simd_store(vtb[14], (float *)(LocalTB + (ik + 7) * 32));
                simd_store(vtb[15], (float *)(LocalTB + 32 * LocalbK + (ik + 7) * 32));
            }

            // 根据是否是最后一个 K 维度的分块进行不同的处理
            if (idK < nK - 1) {
                for (ik = 0; ik + 32 - 1 < LocalbK; ik += 32) {
                    matmul_set_flushing_output(handleMMA, false);
                    matmul_set_output_row_offset(handleMMA, MatmulEnableOutputRowOffset, 0);
                    // 加载权重
                    matmul_load_weight(handleMMA, LocalTB + ik * LocalbN / 2, MatmulK32, MatmulN32);
                    // 执行矩阵乘法计算
                    matmul_compute(handleMMA, pCA + ik, LocalbM, MatmulK32, LocalbK / REDUITEM - 1);

                    matmul_set_flushing_output(handleMMA, false);
                    matmul_set_output_row_offset(handleMMA, MatmulEnableOutputRowOffset, LocalbM);
                    matmul_load_weight(handleMMA,
                                       LocalTB + LocalbN * LocalbK / 2 + ik * LocalbN / 2,
                                       MatmulK32, MatmulN32);
                    matmul_compute(handleMMA, pCA + ik, LocalbM, MatmulK32, LocalbK / REDUITEM - 1);
                }
            } else {
                for (ik = 0; ik + 64 - 1 < LocalbK; ik += 32) {
                    matmul_set_flushing_output(handleMMA, false);
                    matmul_set_output_row_offset(handleMMA, MatmulEnableOutputRowOffset, 0);
                    matmul_load_weight(handleMMA, LocalTB + ik * LocalbN / 2, MatmulK32, MatmulN32);
                    matmul_compute(handleMMA, pCA + ik, LocalbM, MatmulK32, LocalbK / REDUITEM - 1);

                    matmul_set_flushing_output(handleMMA, false);
                    matmul_set_output_row_offset(handleMMA, MatmulEnableOutputRowOffset, LocalbM);
                    matmul_load_weight(handleMMA,
                                       LocalTB + LocalbN * LocalbK / 2 + ik * LocalbN / 2,
                                       MatmulK32, MatmulN32);
                    matmul_compute(handleMMA, pCA + ik, LocalbM, MatmulK32, LocalbK / REDUITEM - 1);
                }
                matmul_set_flushing_output(handleMMA, false);
                matmul_set_output_row_offset(handleMMA, MatmulEnableOutputRowOffset, 0);
                matmul_load_weight(handleMMA, LocalTB + ik * LocalbN / 2, MatmulK32, MatmulN32);
                matmul_compute(handleMMA, pCA + ik, LocalbM, MatmulK32, LocalbK / REDUITEM - 1);

                matmul_set_flushing_output(handleMMA, true);
                matmul_set_output_row_offset(handleMMA, MatmulEnableOutputRowOffset, LocalbM);
                matmul_load_weight(handleMMA, LocalTB + LocalbN * LocalbK / 2 + ik * LocalbN / 2,
                                   MatmulK32, MatmulN32);
                matmul_compute(handleMMA, pCA + ik, LocalbM, MatmulK32, LocalbK / REDUITEM - 1);
                // 存储矩阵乘法结果
                matmul_store(handleMMA, LocalCompC, LocalbM * 2, MatmulN32);
            }
        }   // loop idk
        // 等待矩阵乘法操作完成
        matmul_wait(handleMMA);

        // 对结果 C 进行置换操作
        for (im = 0; im < LocalbM; im++) {
            simd_load(vtb[0], LocalCompC + im * 32);
            simd_load(vtb[1], LocalCompC + im * 32 + 16);
            simd_load(vtb[2], LocalCompC + (im + LocalbM) * 32);
            simd_load(vtb[3], LocalCompC + (im + LocalbM) * 32 + 16);
            simd_store(vtb[0], tempC + im * 64);
            simd_store(vtb[1], tempC + im * 64 + 16);
            simd_store(vtb[2], tempC + im * 64 + 32);
            simd_store(vtb[3], tempC + im * 64 + 48);
        }

        // 在存储结果之前，将计算结果进行类型转换
        batch_S2H(tempC, (half *)LocalC, 4096);

        // 将结果复制回矩阵 C 的对应位置
        memcpy_stride(pC + idM * bM * ldc + idN * bN, LocalC, LenC, BsizeC, {(size_t)StrideC, 0});
    }   // Loop idx

    // 性能采样
    perf_stop();
    // 打印出性能采样结果
    perf_print();

    // 释放分配的内存
    free(LocalCompA);
    free(LocalCompB);
    free(LocalC);
    free(LocalCompC);
}

int main()
```

## 16.4.2.3 广播

### 16.4.2.3.1 实现原理

本章主要介绍如何在矩阵乘法单元的基础上，使用广播优化策略，提升算子性能。

#### 优化原理

针对多个计算核心同时读取 Global 存储空间中同一块数据的场景，可以使用广播（Broadcast）进行优化：一个计算核心将 Global 存储上的数据读入自己的 SPM 存储空间后，通过广播的方式，将数据同时传输到多个计算核心的 SPM 存储空间。

在 GEMM 计算过程中，根据实际情况，对矩阵 A 和 B 进行或列划分，并将划分后的数据分配到不同的计算核心，进行矩阵乘计算。如果各计算核心均独立地从 Global 存储空间读取相同的数据，会导致资源的浪费。因此，一个计算核心可以先将 Global 存储中的数据读取到自身的 SPM 存储空间，再通过广播策略，同步将该数据传输到多个计算核心的 SPM 存储空间，从而优化数据读取流程，减少不必要的资源浪费，提升整体的计算效率与资源利用率。

SDAA C 中广播接口使用方法，可参考 Broadcast 数据搬运章节。

> 图1 广播的实现原理

#### 代码逻辑

广播通过 `tecoKernelGemmBroadcastImpl` 函数实现，该函数在矩阵乘法单元的基础上，使用广播进行性能优化。

1. 根据 `threadIdx` 指定矩阵的行索引 `rid` 和列索引 `cid`。

2. 从 `pGemm` 结构体中获取矩阵 A 和 B 的维度信息。

3. 分配 SPM 存储空间，包括 `LocalCompA`、`LocalCompB`、`LocalTB`、`LocalC`、`LocalCompC` 和 `tempC`。

4. 配置 `BroadcastHandle` 接口参数。

5. 计算用于广播的线程索引 `typeAThreads` 和 `typeBThreads`，并创建线程组 `col_thread_group` 和 `row_thread_group` 以及相应的广播句柄 `col_handle` 和 `row_handle`。

6. 调用 `broadcast_async` 函数，分别对矩阵 A 和 B 执行非阻塞型广播操作。

   对于矩阵 A：

   - `pDA` 是目标缓冲区的指针，即需要将数据广播到的本地缓冲区位置，用于暂存广播来的数据。
   - `(void *)pA` 是数据源指针，即要广播的数据所在的原始内存位置。
   - `BsizeA` 表示每次广播的数据大小，确定每次传输的数据量规模。
   - `typeAThreads` 用于指定参与此次列广播操作的线程信息。
   - `BroadcastGlobalToSpm` 表示广播方向的参数。
   - `col_handle` 用于管理矩阵 A 列广播操作的句柄，用于控制整个广播过程。

   对于矩阵 B：参数含义与矩阵 A 的类似，只是对应的是行广播相关的参数。

   - `pDB` 是目标缓冲区的指针，即需要将数据广播到的本地缓冲区位置，用于暂存广播来的数据。
   - `(void *)pB` 是数据源指针，即要广播的数据所在的原始内存位置。
   - `BsizeB` 表示每次广播的数据大小，确定每次传输的数据量规模。
   - `typeBThreads` 用于指定参与此次行广播操作的线程信息。
   - `BroadcastGlobalToSpm` 表示广播方向的参数。
   - `row_handle` 用于管理矩阵 B 行广播操作的句柄，用于控制整个广播过程。

7. 执行矩阵乘法计算，存储计算结果。

示例代码如下：

```cpp
__global__ void tecoKernelGemmBroadcastImpl(GEMMArgs pGemm)
{
    // 计算线程行索引和列索引
    const int rid = threadIdx / 8;
    const int cid = threadIdx % 8;

    // 从 GEMMArgs 结构体中获取矩阵的维度信息
    const int M = pGemm.m;
    const int N = pGemm.n;
    const int K = pGemm.k;

    // 从 GEMMArgs 结构体中获取矩阵 A、B、C 的存储步长
    const int lda = pGemm.lda;
    const int ldb = pGemm.ldb;
    const int ldc = pGemm.ldc;

    // 获取矩阵 A、B、C 的指针
    const half *A = (const half *)pGemm.A;
    const half *B = (const half *)pGemm.B;
    half *C = (half *)pGemm.C;

    // 外部分块的大小
    const int bM = pGemm.bM;
    const int bN = pGemm.bN;
    const int bK = pGemm.bK;

    // 外部分块的数量
    int nM = M / bM;
    int nN = N / bN;
    int nK = K / bK;

    // 内部块的大小
    int LocalbM = bM / 8;
    int LocalbN = bN / 4;
    int LocalbK = bK;

    // 内部块的位置
    const half *pA = A + cid * LocalbM * lda;
    const half *pB = B + rid * LocalbN;
    half *pC = C + cid * LocalbM * ldc + rid * LocalbN;

    // 指向当前操作元素的指针
    const half *pCurrA;
    const half *pCurrB;
    half *pCurrC;

    // 计算存储数据所需的长度和步长
    const int LenA = LocalbM * LocalbK * sizeof(half);
    const int BsizeA = LocalbK * sizeof(half);
    const int StrideA = (lda - LocalbK) * sizeof(half);

    const int LenB = LocalbK * LocalbN * sizeof(half);
    const int BsizeB = LocalbN * sizeof(half);
    const int StrideB = (ldb - LocalbN) * sizeof(half);

    const int LenC = LocalbM * LocalbN * sizeof(half);
    const int BsizeC = LocalbN * sizeof(half);
    const int StrideC = (ldc - LocalbN) * sizeof(half);
    // 计算所需的SPM存储空间大小
    int spm_size = LenA + 2 * LenB + LenC + LenC / sizeof(half) * sizeof(float) * 2;

    // 分配本SPM存储空间
    half *LocalCompA = (half *)malloc(LenA);
    half *LocalCompB = (half *)malloc(LenB * 2);
    half *LocalTB = LocalCompB + LocalbK * LocalbN;
    half *LocalC = (half *)malloc(LenC);
    // 存储中间结果
    float *LocalCompC = (float *)malloc(LocalbM * LocalbN * sizeof(float) * 2);
    float *tempC = LocalCompC + LocalbM * LocalbN;
    half *pCA = LocalCompA;
    half *pCB = LocalCompB;
    int im, in, ik;
    int idM, idN, idK;

    SIMDType vtb[SIMDSIZE];

    // 开始性能采样
    perf_start();

    // 初始化矩阵乘法句柄
    MatmulHandle handleMMA;
    matmul_init(handleMMA, MatmulHalfToFloat);

    // 计算用于广播的线程索引，列广播
    int typeAThreads = (3 - cid % 4) * 8 + cid;
    // 计算用于广播的线程索引，行广播
    int typeBThreads = rid * 8 + rid;

    // 用于列广播的线程掩码
    uint64_t mask = 0;
    mask = 0x01010101 << cid;
    ThreadGroup col_thread_group(mask);

    // 用于行广播的线程掩码
    mask = 0xFF << (8 * rid);
    ThreadGroup row_thread_group(mask);

    // 列广播的句柄
    BroadcastHandle col_handle(&col_thread_group);
    // 行广播的句柄
    BroadcastHandle row_handle(&row_thread_group);

    // 遍历矩阵的分块
    for (int idx = 0; idx < nM * nN; ++idx) {
        // 计算当前分块的行索引和列索引
        idM = idx / nN;
        idN = idx % nN;
        for (idK = 0; idK < nK; ++idK) {
            // 获取矩阵 A 和 B 中当前分块的位置
            pCurrA = pA + idM * bM * lda + idK * bK;   // A[idM][idK]
            pCurrB = pB + idK * bK * ldb + idN * bN;   // B[idK][idN]

            // 等待矩阵乘法句柄加载输入
            matmul_wait_loading_input(handleMMA);
            // 同步列广播的线程组
            sync_threads(col_thread_group);
            // 执行列广播操作，将数据从Global内存广播到SPM内存
            broadcast(pCA, (void *)pCurrA, LenA, BsizeA, {0, (size_t)StrideA}, typeAThreads, BroadcastGlobalToSpm, col_handle);

            // 等待矩阵乘法句柄加载权重
            matmul_wait_loading_weight(handleMMA);
            // 同步行广播的线程组
            sync_threads(row_thread_group);
            // 执行广播操作，将数据从Global内存广播到SPM内存
            broadcast(pCB, (void *)pCurrB, LenB, BsizeB, {0, (size_t)StrideB}, typeBThreads, BroadcastGlobalToSpm, row_handle);

            // 使用 SIMD 加载和存储数据，对 B 矩阵进行转置存储
            for (ik = 0; ik < LocalbK; ik += 8) {
                simd_load(vtb[0], (float *)(pCB + ik * LocalbN));
                simd_load(vtb[1], (float *)(pCB + ik * LocalbN + 32));
                simd_load(vtb[2], (float *)(pCB + (ik + 1) * LocalbN));
                simd_load(vtb[3], (float *)(pCB + (ik + 1) * LocalbN + 32));
                simd_load(vtb[4], (float *)(pCB + (ik + 2) * LocalbN));
                simd_load(vtb[5], (float *)(pCB + (ik + 2) * LocalbN + 32));
                simd_load(vtb[6], (float *)(pCB + (ik + 3) * LocalbN));
                simd_load(vtb[7], (float *)(pCB + (ik + 3) * LocalbN + 32));
                simd_load(vtb[8], (float *)(pCB + (ik + 4) * LocalbN));
                simd_load(vtb[9], (float *)(pCB + (ik + 4) * LocalbN + 32));
                simd_load(vtb[10], (float *)(pCB + (ik + 5) * LocalbN));
                simd_load(vtb[11], (float *)(pCB + (ik + 5) * LocalbN + 32));
                simd_load(vtb[12], (float *)(pCB + (ik + 6) * LocalbN));
                simd_load(vtb[13], (float *)(pCB + (ik + 6) * LocalbN + 32));
                simd_load(vtb[14], (float *)(pCB + (ik + 7) * LocalbN));
                simd_load(vtb[15], (float *)(pCB + (ik + 7) * LocalbN + 32));
                simd_store(vtb[0], (float *)(LocalTB + ik * 32));
                simd_store(vtb[1], (float *)(LocalTB + 32 * LocalbK + ik * 32));
                simd_store(vtb[2], (float *)(LocalTB + (ik + 1) * 32));
                simd_store(vtb[3], (float *)(LocalTB + 32 * LocalbK + (ik + 1) * 32));
                simd_store(vtb[4], (float *)(LocalTB + (ik + 2) * 32));
                simd_store(vtb[5], (float *)(LocalTB + 32 * LocalbK + (ik + 2) * 32));
                simd_store(vtb[6], (float *)(LocalTB + (ik + 3) * 32));
                simd_store(vtb[7], (float *)(LocalTB + 32 * LocalbK + (ik + 3) * 32));
                simd_store(vtb[8], (float *)(LocalTB + (ik + 4) * 32));
                simd_store(vtb[9], (float *)(LocalTB + 32 * LocalbK + (ik + 4) * 32));
                simd_store(vtb[10], (float *)(LocalTB + (ik + 5) * 32));
                simd_store(vtb[11], (float *)(LocalTB + 32 * LocalbK + (ik + 5) * 32));
                simd_store(vtb[12], (float *)(LocalTB + (ik + 6) * 32));
                simd_store(vtb[13], (float *)(LocalTB + 32 * LocalbK + (ik + 6) * 32));
                simd_store(vtb[14], (float *)(LocalTB + (ik + 7) * 32));
                simd_store(vtb[15], (float *)(LocalTB + 32 * LocalbK + (ik + 7) * 32));
            }

            // 根据是否是最后一个 K 维度的分块进行不同的处理
            if (idK < nK - 1) {
                for (ik = 0; ik + 32 - 1 < LocalbK; ik += 32) {
                    // 设置矩阵乘法句柄
                    matmul_set_flushing_output(handleMMA, false);
                    // 设置矩阵乘法句柄的输出行偏移
                    matmul_set_output_row_offset(handleMMA, MatmulEnableOutputRowOffset, 0);
                    // 加载权重到矩阵乘法句柄
                    matmul_load_weight(handleMMA, LocalTB + ik * LocalbN / 2, MatmulK32, MatmulN32);
                    // 执行矩阵乘法计算
                    matmul_compute(handleMMA, pCA + ik, LocalbM, MatmulK32, LocalbK / REDUITEM - 1);

                    matmul_set_flushing_output(handleMMA, false);
                    matmul_set_output_row_offset(handleMMA, MatmulEnableOutputRowOffset, LocalbM);
                    matmul_load_weight(handleMMA,
                                       LocalTB + LocalbN * LocalbK / 2 + ik * LocalbN / 2,
                                       MatmulK32, MatmulN32);
                    matmul_compute(handleMMA, pCA + ik, LocalbM, MatmulK32, LocalbK / REDUITEM - 1);
                }
            } else {
                for (ik = 0; ik + 64 - 1 < LocalbK; ik += 32) {
                    matmul_set_flushing_output(handleMMA, false);
                    matmul_set_output_row_offset(handleMMA, MatmulEnableOutputRowOffset, 0);
                    matmul_load_weight(handleMMA, LocalTB + ik * LocalbN / 2, MatmulK32, MatmulN32);
                    matmul_compute(handleMMA, pCA + ik, LocalbM, MatmulK32, LocalbK / REDUITEM - 1);

                    matmul_set_flushing_output(handleMMA, false);
                    matmul_set_output_row_offset(handleMMA, MatmulEnableOutputRowOffset, LocalbM);
                    matmul_load_weight(handleMMA,
                                       LocalTB + LocalbN * LocalbK / 2 + ik * LocalbN / 2,
                                       MatmulK32, MatmulN32);
                    matmul_compute(handleMMA, pCA + ik, LocalbM, MatmulK32, LocalbK / REDUITEM - 1);
                }
                matmul_set_flushing_output(handleMMA, false);
                matmul_set_output_row_offset(handleMMA, MatmulEnableOutputRowOffset, 0);
                matmul_load_weight(handleMMA, LocalTB + ik * LocalbN / 2, MatmulK32, MatmulN32);
                matmul_compute(handleMMA, pCA + ik, LocalbM, MatmulK32, LocalbK / REDUITEM - 1);

                matmul_set_flushing_output(handleMMA, true);
                matmul_set_output_row_offset(handleMMA, MatmulEnableOutputRowOffset, LocalbM);
                matmul_load_weight(handleMMA, LocalTB + LocalbN * LocalbK / 2 + ik * LocalbN / 2,
                                   MatmulK32, MatmulN32);
                matmul_compute(handleMMA, pCA + ik, LocalbM, MatmulK32, LocalbK / REDUITEM - 1);
                // 存储矩阵乘法结果
                matmul_store(handleMMA, LocalCompC, LocalbM * 2, MatmulN32);
            }
        }   // loop idk
        // 等待矩阵乘法操作完成
        matmul_wait(handleMMA);

        // 对结果 C 进行置换操作
        for (im = 0; im < LocalbM; im++) {
            simd_load(vtb[0], LocalCompC + im * 32);
            simd_load(vtb[1], LocalCompC + im * 32 + 16);
            simd_load(vtb[2], LocalCompC + (im + LocalbM) * 32);
            simd_load(vtb[3], LocalCompC + (im + LocalbM) * 32 + 16);
            simd_store(vtb[0], tempC + im * 64);
            simd_store(vtb[1], tempC + im * 64 + 16);
            simd_store(vtb[2], tempC + im * 64 + 32);
            simd_store(vtb[3], tempC + im * 64 + 48);
        }

        // 在存储结果之前，将计算结果进行类型转换
        batch_S2H(tempC, (half *)LocalC, 4096);
        // 将结果复制回矩阵 C 的对应位置
        memcpy_stride(pC + idM * bM * ldc + idN * bN, LocalC, LenC, BsizeC, {(size_t)StrideC, 0});
    }   // Loop idx

    // 性能采样
    perf_stop();
    // 打印出性能采样结果
    perf_print();

    // 释放分配的内存
    free(LocalCompA);
    free(LocalCompB);
    free(LocalC);
    free(LocalCompC);
}
```

```cpp
    // 每次循环处理完一组后，指针移动 128 个元素位置继续处理下一组
    for (i = 0; i <= len - 128; i += 128) {

        simd_load(vx0, x);
        simd_load(vx1, x + 16);
        simd_load(vx2, x + 32);
        simd_load(vx3, x + 48);

        vy4 = simd_cvt_f2h(vx4);

        simd_store(vy1, y + 16);
        simd_store(vy2, y + 32);
        simd_store(vy3, y + 48);
        simd_store(vy4, y + 64);
        simd_store(vy5, y + 80);
        simd_store(vy6, y + 96);
        simd_store(vy7, y + 112);

        x += 128;

    }

    for (; i <= len - 64; i += 64) {

// 根据长度和对齐情况调用 align_batch_function 进行浮点数到半精度浮点数的转换
__device__ void batch_S2H(float* x, half* y, int len)
{
    if (len < 32) {
        for (int i = 0; i < len; i++) {
            y[i] = x[i];
        }
        return;
    }
    const int align_n1 = sizeof(floatv16);
    const int align_n2 = sizeof(halfv16);
    if ((((size_t)x & (align_n1 - 1)) == 0) && ((((size_t)y) & (align_n2 - 1)) == 0)) {
        align_batch_function((float*)x, y, len);
        return;
    } else {
        int padding_x = ((size_t)(x) & (align_n1 - 1)) / sizeof(float);
        int padding_y = ((size_t)(y) & (align_n2 - 1)) / sizeof(half);
        if (padding_x == padding_y) {
            for (int i = 0; i < 16 - padding_x; i++) {
                y[i] = x[i];
            }
            align_batch_function(x + 16 - padding_x, y + 16 - padding_y, len - 16 + padding_x);
            return;
        } else {
            align_batch_function((float*)x, y, len);
            return;
        }
    }
}
```
