## 13.3.2.4 Broadcast 数据搬运

### 13.3.2.4.1 SPE ID 相关

本文主要介绍使用线程组进行 Broadcast 数据搬运的性能优化策略和示例。

#### 优化策略

使用线程组进行 Broadcast 数据搬运时，当线程组内的所有 SPE 满足以下任意一种条件时，性能较好：

- 线程组内所有 SPE 刚好能被划分为：{0~7}，{8~15}，{16~23}，{24~31} 中的一组或多组。
- 线程组内所有 SPE 刚好能被划分为：{0, 8, 16, 24}，{1, 9, 17, 25}，{2, 10, 18, 26}，{3, 11, 19, 27}，{4, 12, 20, 28}，{5, 13, 21, 29}，{6, 14, 22, 30}，{7, 15, 23, 31} 中的一组或多组。

例如：对包含 8 个 SPE 的线程组进行广播时，线程组 A={0, 1, 2, 3, 4, 5, 6, 7} 或 B={0, 1, 8, 9, 16, 17, 24, 25} 的性能要优于线程组 C={0, 1, 9, 10, 18, 19, 27, 28}。

#### 优化示例

对使用线程组 A={0, 1, 2, 3, 4, 5, 6, 7}、B={0, 1, 8, 9, 16, 17, 24, 25} 和 C={0, 1, 9, 10, 18, 19, 27, 28} 进行阻塞型广播的执行周期进行测试并比较。

每个测试均调用 1000 次 broadcast（详情请参考《SDAA C 编程指南》的 broadcast 老接口章节）并计算平均值以消除初次调用时 cache 影响。

#### 测试环境

| TecoDriver 和 TecoToolKit 版本 | 计算核心频率 |
|--------------------------------|--------------|
| v1.2.0 | 2.36GHz |

#### 测试代码

```c
#include <sdaa_perf.h>
using namespace sdaa;

__local__ int src;
__local__ int dst;

__device__ void broadcast_perf(ThreadGroup &tg)
{
        if (threadIdx != 0 && !thread_group_is_included(tg, threadIdx)) return;
        // data init
        BroadcastHandle handle(&tg);
        src = 0;
        dst = 0;
        if (threadIdx == 0) src = 100;
        sync_threads(tg);
        // start broadcast
        perf_start();
        for (int i = 0; i < 1000; i++) {
            broadcast(&dst, &src, sizeof(int), 0, {0, 0}, 0, BroadcastSpmToSpm, handle);
        }
        perf_stop();
        perf_print();
}

// broadcast to 0, 1, 2, 3, 4, 5, 6, 7
__global__ void test_a()
{
        unsigned long thread_group[8] = {0, 1, 2, 3, 4, 5, 6, 7};
        ThreadGroup tg(thread_group, 8);
        broadcast_perf(tg);
}

// broadcast to 0, 1, 8, 9, 16, 17, 24, 25
__global__ void test_b()
{
        unsigned long thread_group[8] = {0, 1, 8, 9, 16, 17, 24, 25};
        ThreadGroup tg(thread_group, 8);
        broadcast_perf(tg);
}

// broadcast to 0, 1, 9, 10, 18, 19, 27, 28
__global__ void test_c()
{
        unsigned long thread_group[8] = {0, 1, 9, 10, 18, 19, 27, 28};
        ThreadGroup tg(thread_group, 8);
        broadcast_perf(tg);
}
int main()
{
        sdaaSetDevice(2);
        test_a<<<1>>>();
        sdaaDeviceSynchronize();
        test_b<<<1>>>();
        sdaaDeviceSynchronize();
        test_c<<<1>>>();
        sdaaDeviceSynchronize();
        return 0;
}
```

#### 优化效果

| 线程组中所有 SPE 的 ID | 平均每次 Broadcast 操作的执行周期（拍） |
|------------------------|-----------------------------------------|
| 0, 1, 2, 3, 4, 5, 6, 7 | 266 |
| 0, 1, 8, 9, 16, 17, 24, 25 | 322 |
| 0, 1, 9, 10, 18, 19, 27, 28 | 1362 |

在本示例中，满足优化策略的条件时使用线程组进行阻塞型 Broadcast 数据搬运的性能，是不满足条件时的约 5.12 倍。

### 13.3.2.4.2 非阻塞型 Broadcast 数据搬运

本文主要介绍非阻塞型 Broadcast 数据搬运的性能优化策略和示例。

#### 优化策略

如果搬运的数据与当前程序的后续操作不存在数据依赖时，使用非阻塞型接口可以更好的进行流水排布，相对阻塞式接口性能更好。

#### 优化示例

在设备端使用阻塞型广播接口将 SPE0 中的数据广播到 SPE0~SPE7 内然后执行与数据搬运无数据依赖的计算，与使用非阻塞型数据搬运接口同步执行将 SPE0 中的数据广播到 SPE0~SPE7 的数据搬运操作和与数据搬运无数据依赖的计算进行性能测试并比较。

每个测试均调用 10000 次数据搬运和与搬运数据无依赖的计算操作，从而计算平均值以消除初次调用时 cache 影响。

#### 测试环境

| TecoDriver 和 TecoToolKit 版本 | 计算核心频率 | 数据搬运量 |
|--------------------------------|--------------|------------|
| v1.2.0 | 2.36GHz | 10KB |

#### 测试代码

```c
#include<sdaa_perf.h>
using namespace sdaa;

#define SIZE 2560
#define SIZE_CAL 256
#define TIME 10000

__device__ __attribute__((noinline)) void test_sync(int *dst, int *bc_dst, int *bc_src, int *d_cal)
{
        // 设置线程组 SPE0~SPE7
        ThreadGroup thread_group(0xff);
        BroadcastHandle handle(&thread_group);
        // 初始化目的地址数据
        if (thread_group_is_included(thread_group, threadIdx)) {
            memset(bc_dst, 0, SIZE * sizeof(int));
        }

        intv16 cal_src;
        intv16 add = 1;
        // 执行SPE0~SPE7同步
        sync_threads(thread_group);

        if (thread_group_is_included(thread_group, threadIdx)) {
            // 统计性能数据
            perf_start();

            for (int time = 0; time < TIME; time++) {
                broadcast(bc_dst, bc_src, SIZE * sizeof(int), 0, {0, 0}, 0, BroadcastSpmToSpm, handle);

                // 异步进行与数据搬运无数据依赖的计算
                for (int idx = 0; idx < SIZE_CAL; idx += 16) {
                    // 使用向量进行计算
                    simd_load(cal_src, d_cal + (SIZE_CAL * threadIdx) + idx);
                    intv16 tmp = cal_src + add;
                    simd_store(tmp, d_cal + (SIZE_CAL * threadIdx) + idx);
                }
            }
            // 统计性能数据
            perf_stop();
        }

        // 将广播数据拷贝到SPE0~SPE7的目的地址
        if (thread_group_is_included(thread_group, threadIdx)) {
            memcpy(dst + threadIdx * SIZE, bc_dst, SIZE * sizeof(int));
        }
}

__device__ __attribute__((noinline)) void test_async(int *dst, int *bc_dst, int *bc_src, int *d_cal)
{
        // 设置线程组 SPE0~SPE7
        ThreadGroup thread_group(0xff);
        BroadcastHandle handle(&thread_group);
        // 初始化目的地址数据
        if (thread_group_is_included(thread_group, threadIdx)) {
            memset(bc_dst, 0, SIZE * sizeof(int));
        }

        intv16 cal_src;
        intv16 add = 1;
        // 执行SPE0~SPE7同步
        sync_threads(thread_group);

        if (thread_group_is_included(thread_group, threadIdx)) {
            // 统计性能数据
            perf_start();

            for (int time = 0; time < TIME; time++) {
                // 发起非阻塞广播操作
                broadcast_async(bc_dst, bc_src, SIZE * sizeof(int), 0, {0, 0}, 0, BroadcastSpmToSpm, handle);

                // 同步进行与数据搬运无数据依赖的计算
                for (int idx = 0; idx < SIZE_CAL; idx += 16) {
                    // 使用向量进行计算
                    simd_load(cal_src, d_cal + (SIZE_CAL * threadIdx) + idx);
                    intv16 tmp = cal_src + add;
                    simd_store(tmp, d_cal + (SIZE_CAL * threadIdx) + idx);
                }
                // 等待非阻塞广播操作完成
                broadcast_wait(handle);
            }
            // 统计性能数据
            perf_stop();
        }

        // 将广播数据拷贝到SPE0~SPE7的目的地址
        if (thread_group_is_included(thread_group, threadIdx)) {
            memcpy(dst + threadIdx * SIZE, bc_dst, SIZE * sizeof(int));
        }
}

__global__ void test(int *d_sync, int *d_async, int *d_sync_cal, int *d_async_cal)
{
        int *src = (int *)malloc(SIZE * sizeof(int));
        int *dst = (int *)malloc(SIZE * sizeof(int));
        if (threadIdx == 0) {
            for (int i = 0; i < SIZE; i++) {
                src[i] = i;
            }
        }
        // 阻塞接口测试
        test_sync(d_sync, dst, src, d_sync_cal);
        // 非阻塞接口测试
        test_async(d_async, dst, src, d_async_cal);

        // 打印性能数据
        perf_print();

        free(dst);
        free(src);
}

int main()
{
    const int thread_cnt = 8;
    int h_dst_async[SIZE * thread_cnt] = { 0 };
    int h_dst_sync[SIZE * thread_cnt] = { 0 };
    int h_sync_cal[SIZE_CAL * thread_cnt] = { 0 };
    int h_async_cal[SIZE_CAL * thread_cnt] = { 0 };

    for (int i = 0; i < SIZE_CAL * thread_cnt; i++) {
        h_sync_cal[i] = i;
        h_async_cal[i] = i;
    }

    // 设置使用0号SPA
    sdaaSetDevice(0);
    int *d_dst_async = NULL;
    int *d_dst_sync = NULL;
    int *d_sync_cal = NULL;
    int *d_async_cal = NULL;

    // 在Global内存中malloc一段空间
    sdaaMalloc((void**)&d_dst_async, SIZE * thread_cnt * sizeof(int));
    sdaaMalloc((void**)&d_dst_sync, SIZE * thread_cnt * sizeof(int));
    sdaaMalloc((void**)&d_sync_cal, SIZE_CAL * thread_cnt * sizeof(int));
    sdaaMalloc((void**)&d_async_cal, SIZE_CAL * thread_cnt * sizeof(int));

    // 将数据从Host端拷贝到Device端
    sdaaMemcpy(d_dst_async, h_dst_async, SIZE * thread_cnt * sizeof(int), sdaaMemcpyHostToDevice);
    sdaaMemcpy(d_dst_sync, h_dst_sync, SIZE * thread_cnt * sizeof(int), sdaaMemcpyHostToDevice);
    sdaaMemcpy(d_sync_cal, h_sync_cal, SIZE_CAL * thread_cnt * sizeof(int), sdaaMemcpyHostToDevice);
    sdaaMemcpy(d_async_cal, h_async_cal, SIZE_CAL * thread_cnt * sizeof(int), sdaaMemcpyHostToDevice);

    // kernel函数调用
    test<<<1>>>(d_dst_sync, d_dst_async, d_sync_cal, d_async_cal);

    // 等待kernel函数执行完成
    sdaaDeviceSynchronize();
    // 将数据从Device端拷贝到Host端
    sdaaMemcpy(h_dst_async, d_dst_async, SIZE * thread_cnt * sizeof(int), sdaaMemcpyDeviceToHost);
    sdaaMemcpy(h_dst_sync, d_dst_sync, SIZE * thread_cnt * sizeof(int), sdaaMemcpyDeviceToHost);
    sdaaMemcpy(h_sync_cal, d_sync_cal, SIZE_CAL * thread_cnt * sizeof(int), sdaaMemcpyDeviceToHost);
    sdaaMemcpy(h_async_cal, d_async_cal, SIZE_CAL * thread_cnt * sizeof(int), sdaaMemcpyDeviceToHost);

    for (int i = 0; i < SIZE * thread_cnt; i++) {
        if ((h_dst_sync[i] != i % SIZE) || (h_dst_async[i] != i % SIZE)) {
             printf("h_dst_sync[%d] = %d, h_dst_async[%d] = %d\n", i, h_dst_sync[i], i, h_dst_async[i]);
        }
    }

    for (int i = 0; i < SIZE_CAL * thread_cnt; i++) {
        if ((h_sync_cal[i] != i + TIME) || (h_async_cal[i] != i + TIME)) {
             printf("h_sync_cal[%d] = %d, h_async_cal[%d] = %d\n",
                    i, h_sync_cal[i], i, h_async_cal[i]);
        }
    }

    // 释放申请的Global内存
    sdaaFree(d_sync_cal);
    sdaaFree(d_async_cal);
    sdaaFree(d_dst_async);
    sdaaFree(d_dst_sync);
    return 0;
}
```

## 13.3.2.5 原子操作

本文主要介绍原子操作的性能优化策略和示例。

原子操作除 atomic_inc 外，返回旧值的接口相比不返回旧值的接口会多一些操作，导致性能上的差别。

### 优化策略

当不需要返回指定地址上的旧值时，可以使用不返回旧值的原子操作接口提升性能：

- 使用 atomic_add_noret 代替 atomic_add。
- 使用 atomic_sub_noret 代替 atomic_sub。
- 使用 atomic_cas_bool 代替 atomic_cas。

### 优化示例

同一 SPA 上的所有 SPE 分别使用有返回值的原子操作接口和没有返回值的原子操作接口，测试并比较以下操作的执行周期：

- 所有 SPE 均执行 1000 次 atomic_add_noret 和 atomic_add。
- 所有 SPE 均执行 1000 次 atomic_sub_noret 和 atomic_sub。
- 所有 SPE 均执行 1000 次 atomic_cas_bool 和 atomic_cas。

### 测试环境

| TecoDriver 和 TecoToolKit 版本 | 计算核心频率 |
|--------------------------------|--------------|
| v1.2.0 | 2.36GHz |

### 测试代码

```c
#include <sdaa_perf.h>
#include <sdaa_atomic.h>
using namespace sdaa;

__device__ int global_val;

// 使用atomic_add接口
__global__ void test_atomic_add()
{
        // 数据初始化
        const int step = 10;
        if (threadIdx == 0) global_val = 0;
        // 对每个线程执行1000次atomic_add
        sync_threads();
        perf_start();
        for (int i = 0; i < 1000; i++) {
            atomic_add(&global_val, step);
        }
        perf_stop();

        // 检查结果的正确性
        sync_threads();
        if (threadIdx != 0) return;
        if (global_val != 1000 * threadDim * step) {
            printf("Error: atomic operations not working correctly!\n");
        }
        // 输出性能数据
        perf_print();
}

// 使用atomic_add_noret接口
__global__ void test_atomic_add_noret()
{
        // 数据初始化
        const int step = 10;
        if (threadIdx == 0) global_val = 0;
        // 对每个线程执行1000次atomic_add_noret
        sync_threads();
        perf_start();
        for (int i = 0; i < 1000; i++) {
            atomic_add_noret(&global_val, step);
        }
        perf_stop();

        // 检查结果的正确性
```

## 13.3.2.6 矩阵乘

### 13.3.2.6.1 阻塞型矩阵乘

矩阵乘阻塞型接口相比非阻塞型接口具有更少的使用限制和更高的通用性，但相应地性能也会差于非阻塞型接口。当满足一定对齐条件时，矩阵乘阻塞型接口的性能可以提升到接近非阻塞型接口的性能。

#### 优化策略

当阻塞型矩阵乘接口满足一下条件时，性能较好：

k（输入矩阵列数、权重矩阵行数）和 n（权重矩阵列数）均是 32 的整数倍。

当使用阻塞型矩阵乘接口时，预先将矩阵分块为 32 大小的整数倍，可以有效提高矩阵乘接口的性能。

#### 优化示例

单一线程使用阻塞型矩阵乘接口计算 SPM 上矩阵的乘积时，测试以下两种情况的执行周期并比较：

- k 和 n 均为 32 的整数倍（m=128，k=64，n=32）。
- k 和 n 不是 32 的整数倍（m=128，k=60，n=30）。

#### 测试环境

| TecoDriver 和 TecoToolKit 版本 | 计算核心频率 |
|--------------------------------|--------------|
| v1.2.0 | 2.36GHz |

#### 测试代码

```c
#include <sdaa_perf.h>
#include <sdaa_matmul.h>
using namespace sdaa;

__device__ void matmul_perf(half *output, half *input, half *weight,
                                unsigned int m, unsigned int k, unsigned int n)
{
        // 获取性能数据起始值
        perf_start();
        for (int i = 0; i < 1000; i++) {
            matmul(MatmulHalfToHalf, output, input, weight, m, k, n);
        }
        // 获取性能数据结束值
        perf_stop();
        // 输出性能数据
        perf_print();
}

__device__ void init_matrix(half *mat, unsigned int m, unsigned int n)
{
        // 初始化矩阵数组
        for (unsigned int i = 0; i < m; i++) {
            for (unsigned int j = 0; j < n; j++) {
                mat[i * n + j] = (half) m / (half) n;
            }
        }
}

__global__ void test_aligned()
{
        if (threadIdx != 0) return;
        const unsigned int m = 128, k = 64, n = 32;
        half *input = (half *)malloc(m * k * sizeof(half));
        half *weight = (half *)malloc(k * n * sizeof(half));
        half *output = (half *)malloc(m * n * sizeof(half));
        init_matrix(input, m, k);
        init_matrix(weight, k, n);
        matmul_perf(output, input, weight, m, k, n);
        free(output);
        free(weight);
        free(input);
}

__global__ void test_unaligned()
{
    if (threadIdx != 0) return;
    const unsigned int m = 128, k = 60, n = 30;
    half *input = (half *)malloc(m * k * sizeof(half));
    half *weight = (half *)malloc(k * n * sizeof(half));
    half *output = (half *)malloc(m * n * sizeof(half));
    init_matrix(input, m, k);
    init_matrix(weight, k, n);
    matmul_perf(output, input, weight, m, k, n);
    free(output);
    free(weight);
    free(input);
}

int main()
{
    // 设置使用0号SPA
    sdaaSetDevice(2);
    // kernel函数调用
    test_aligned<<<1>>>();
    // 等待kernel函数执行完成
    sdaaDeviceSynchronize();
    // kernel函数调用
    test_unaligned<<<1>>>();
    // 等待kernel函数执行完成
    sdaaDeviceSynchronize();
    return 0;
}
```

#### 优化效果

| 矩阵维度大小 | 平均每次阻塞型矩阵乘接口的执行周期（拍） |
|--------------|------------------------------------------|
| m=128，k=64，n=32 | 26993 |
| m=128，k=60，n=30 | 146818 |

在本示例中，k 和 n 非 32 整数倍（k=60，n=30）的矩阵乘数据量更小，但执行周期更长。k 和 n 是 32 整数倍（k=64，n=32）的矩阵乘虽然多了一些数据，但性能却是非 32 整数倍的矩阵乘的 5.43 倍。出现这个性能差距的原因是非 32 整数倍的矩阵需要重新切分组装数据，耗时较多。

### 13.3.2.6.2 非阻塞型矩阵乘

本文主要介绍非阻塞型矩阵乘的性能优化策略和示例。

#### 优化策略

如果搬运的数据与当前程序的后续操作不存在数据依赖时，使用非阻塞式接口可以更好的进行流水排布，相对阻塞式接口性能更好。

#### 优化示例

在设备端测试以下情况的数据计算总执行周期并比较：

- 使用阻塞型矩阵乘接口：先执行矩阵乘，再执行与矩阵乘无数据依赖的计算操作。
- 使用非阻塞型矩阵乘接口：同步执行矩阵乘和与矩阵乘无数据依赖的计算操作。

每个测试均调用 10000 次矩阵乘操作和与矩阵乘无数据依赖的计算操作，从而计算平均值以消除初次调用时 cache 影响。

#### 测试环境

| TecoDriver 和 TecoToolKit 版本 | 计算核心频率 | 输入矩阵大小 | 权重矩阵大小 |
|--------------------------------|--------------|--------------|--------------|
| v1.2.0 | 2.36GHz | 128 × 32 | 32 × 32 |

#### 测试代码

```c
#include <sdaa_matmul.h>
#include <sdaa_perf.h>
using namespace sdaa;

#define WEIGHT_ROW 32
#define WEIGHT_COL 32
#define INTPUT_ROW 128
#define INTPUT_COL 32
#define OUTPUT_ROW 128
#define OUTPUT_COL 32
#define SIZE_CAL 256
#define TIME 10000

#define INTPUT_SIZE (INTPUT_ROW * INTPUT_COL * sizeof(short))
#define WEIGHT_SIZE (WEIGHT_ROW * WEIGHT_COL * sizeof(short))
#define OUTPUT_SIZE (OUTPUT_ROW * OUTPUT_COL * sizeof(short))

__device__ void test_sync(short *input, short *weight, short *output, int *cal)
{
        short *ibuffer = (short *)malloc(INTPUT_SIZE);
        short *wbuffer = (short *)malloc(WEIGHT_SIZE);
        short *obuffer = (short *)malloc(OUTPUT_SIZE);

        memcpy(ibuffer, input, INTPUT_SIZE);
        memcpy(wbuffer, weight, WEIGHT_SIZE);
        memcpy(obuffer, output, OUTPUT_SIZE);

        intv16 cal_src;
        intv16 add = 1;
        // 统计性能数据
        perf_start();

        for (int time = 0; time < TIME; time++) {
            matmul(MatmulShortToShort, obuffer, ibuffer, wbuffer, INTPUT_ROW, WEIGHT_ROW, OUTPUT_COL);
            // 异步进行与数据搬运无数据依赖的计算
            for (int idx = 0; idx < SIZE_CAL; idx += 16) {
                // 使用向量进行计算
                simd_load(cal_src, cal + idx);
                intv16 tmp = cal_src + add;
                simd_store(tmp, cal + idx);
            }
        }
        // 统计性能数据
        perf_stop();

        memcpy(output, obuffer, OUTPUT_SIZE);

        free(ibuffer);
        free(wbuffer);
        free(obuffer);
}

__device__ void test_async(short *input, short *weight, short *output, int *cal)
{
    short *ibuffer = (short *)malloc(INTPUT_SIZE);
    short *wbuffer = (short *)malloc(WEIGHT_SIZE);
    short *obuffer = (short *)malloc(OUTPUT_SIZE);

    memcpy(ibuffer, input, INTPUT_SIZE);
    memcpy(wbuffer, weight, WEIGHT_SIZE);
    memcpy(obuffer, output, OUTPUT_SIZE);

    MatmulHandle handle;
    intv16 cal_src;
    intv16 add = 1;
    // 统计性能数据
    perf_start();

    for (int time = 0; time < TIME; time++) {
        // 初始化handle
        matmul_init(handle, MatmulShortToShort);
        // 加载权重矩阵
        matmul_load_weight(handle, wbuffer, MatmulK32, MatmulN32);
        // 加载输入矩阵并与权重矩阵进行矩阵乘运算
        matmul_compute(handle, ibuffer, INTPUT_ROW, MatmulK32);
        // 将矩阵乘运算结果写回到指定地址
        matmul_store(handle, obuffer, OUTPUT_ROW, MatmulN32);

        for (int idx = 0; idx < SIZE_CAL; idx += 16) {
            // 使用向量进行计算
            simd_load(cal_src, cal + idx);
            intv16 tmp = cal_src + add;
            simd_store(tmp, cal + idx);
        }
        // 等待矩阵乘写回操作完成
        matmul_wait(handle);
    }
    // 统计性能数据
    perf_stop();

    memcpy(output, obuffer, OUTPUT_SIZE);

    free(ibuffer);
    free(wbuffer);
    free(obuffer);
}

__global__ void test(short *input, short *weight, short *output_sync,
                      short *output_async, int *cal_sync, int *cal_async)
{
    if (threadIdx != 0) {
        return;
    }

    // 测试同步接口
    test_sync(input, weight, output_sync, cal_sync);
    // 测试异步接口
    test_async(input, weight, output_async, cal_async);

    perf_print();
}

// 初始化数组
void SetArrayVal(short *array, int num, short val) {
    for (int i = 0; i < num; i++) {
        array[i] = val;
    }
}

void entry(short *h_intput, short *h_weight, short *h_output_sync,
            short *h_output_async, int *h_cal_sync, int *h_cal_async)
{
    // 设置使用0号SPA
    sdaaSetDevice(0);
    short *dev_intput = NULL;
    short *dev_weight = NULL;
    short *dev_output_sync = NULL;
    short *dev_output_async = NULL;
    int *dev_cal_sync = NULL;
    int *dev_cal_async = NULL;

    // 在Global内存中malloc一段空间
    sdaaMalloc((void **)(&dev_intput), INTPUT_SIZE);
    sdaaMalloc((void **)(&dev_weight), WEIGHT_SIZE);
    sdaaMalloc((void **)(&dev_output_sync), OUTPUT_SIZE);
    sdaaMalloc((void **)(&dev_output_async), OUTPUT_SIZE);
    sdaaMalloc((void **)(&dev_cal_sync), SIZE_CAL * sizeof(int));
    sdaaMalloc((void **)(&dev_cal_async), SIZE_CAL * sizeof(int));

    // 将数据从Host端拷贝到Device端
    sdaaMemcpy(dev_intput, h_intput, INTPUT_SIZE, sdaaMemcpyHostToDevice);
    sdaaMemcpy(dev_weight, h_weight, WEIGHT_SIZE, sdaaMemcpyHostToDevice);
    sdaaMemcpy(dev_output_sync, h_output_sync, OUTPUT_SIZE, sdaaMemcpyHostToDevice);
    sdaaMemcpy(dev_output_async, h_output_async, OUTPUT_SIZE, sdaaMemcpyHostToDevice);
    sdaaMemcpy(dev_cal_sync, h_cal_sync, SIZE_CAL * sizeof(int), sdaaMemcpyHostToDevice);
    sdaaMemcpy(dev_cal_async, h_cal_async, SIZE_CAL * sizeof(int), sdaaMemcpyHostToDevice);

    // kernel函数调用
    test<<<1>>>(dev_intput, dev_weight, dev_output_sync, dev_output_async,
                 dev_cal_sync,    dev_cal_async);
    // 等待kernel函数执行完成
    sdaaDeviceSynchronize();

    // 将数据从Device端拷贝到Host端
    sdaaMemcpy(h_output_sync, dev_output_sync, OUTPUT_SIZE, sdaaMemcpyDeviceToHost);
    sdaaMemcpy(h_output_async, dev_output_async, OUTPUT_SIZE, sdaaMemcpyDeviceToHost);
    sdaaMemcpy(h_cal_sync, dev_cal_sync, SIZE_CAL * sizeof(int), sdaaMemcpyDeviceToHost);
    sdaaMemcpy(h_cal_async, dev_cal_async, SIZE_CAL * sizeof(int), sdaaMemcpyDeviceToHost);

    // 释放申请的Global内存
    sdaaFree(dev_intput);
    sdaaFree(dev_weight);
    sdaaFree(dev_output_sync);
    sdaaFree(dev_output_async);
    sdaaFree(dev_cal_sync);
    sdaaFree(dev_cal_async);
}

int main()
{
    short *h_intput = (short *)malloc(INTPUT_SIZE);
    short *h_weight = (short *)malloc(WEIGHT_SIZE);
    short *h_output_sync = (short *)malloc(OUTPUT_SIZE);
    short *h_output_async = (short *)malloc(OUTPUT_SIZE);

    int *h_cal_sync = (int *)malloc(SIZE_CAL * sizeof(int));
    int *h_cal_async = (int *)malloc(SIZE_CAL * sizeof(int));

    SetArrayVal(h_intput, INTPUT_ROW * INTPUT_COL, 1);
    SetArrayVal(h_weight, WEIGHT_ROW * WEIGHT_COL, 2);
    memset(h_output_sync, 0, OUTPUT_SIZE);
    memset(h_output_async, 0, OUTPUT_SIZE);
    memset(h_cal_sync, 0, SIZE_CAL * sizeof(int));
    memset(h_cal_async, 0, SIZE_CAL * sizeof(int));
    const short REFEROUT = 64;

    entry(h_intput, h_weight, h_output_sync, h_output_async, h_cal_sync, h_cal_async);

    for (int row = 0; row < OUTPUT_ROW; row++) {
        for (int col = 0; col < OUTPUT_COL; col++) {
             if ((h_output_sync[row * OUTPUT_COL + col] != REFEROUT) ||
                 (h_output_async[row * OUTPUT_COL + col] != REFEROUT)) {
                 printf("h_output_sync[%d] = %d, h_output_async[%d] = %d\n",
                         row * OUTPUT_COL + col, h_output_sync[row * OUTPUT_COL + col],
                         row * OUTPUT_COL + col, h_output_async[row * OUTPUT_COL + col]);
             }
        }
    }

    for (int i = 0; i < SIZE_CAL; i++) {
        if ((h_cal_sync[i] != TIME) || (h_cal_async[i] != TIME)) {
            printf("h_cal_sync[%d] = %d, h_cal_async[%d] = %d\n",
                   i, h_cal_sync[i], i, h_cal_async[i]);
        }
    }

    free(h_intput);

    free(h_output_sync);
    free(h_output_async);
    free(h_cal_sync);
    free(h_cal_async);
```

#### 优化策略

在使用 SIMD 接口时，源地址和目的地址的数据均位于 SPM 存储空间，性能更好。

#### 优化示例

使用 simd_load 接口，测试并比较以下两种情况，一个 SPE 执行 1000 次的执行周期：

- 源地址和目的地址的数据均位于 SPM 存储空间。
- 源地址和目的地址的数据均位于 Global 存储空间。

#### 测试环境

| TecoDriver 和 TecoToolKit 版本 | 计算核心频率 |
|--------------------------------|--------------|
| v1.2.0 | 2.36GHz |

#### 测试代码

```c
#include <sdaa_perf.h>
using namespace sdaa;
// 在Global内存空间定义变量
volatile __device__ intv16 glb_vdata;

__device__ void test_simd_load_in_global(int *glb_data)
{
        // 收集性能数据起始值
        perf_start();
        for (int i = 0; i < 1000; i++){
            simd_load(glb_vdata, glb_data);
        }
        // 收集性能数据结束值
        perf_stop();
}

__device__ void test_simd_load_in_spm(int *glb_data)
{
    int *spm_data = (int *)malloc(16 * sizeof(int));
    memcpy(spm_data, glb_data, 16 * sizeof(int));
    volatile intv16 spm_vdata;

    perf_start();
    for (int i = 0; i < 1000; i++) {
        simd_load(spm_vdata, spm_data);
    }
    perf_stop();

    free(spm_data);
}

__global__ void test_func(int *glb_data)
{
    if (threadIdx != 0) return;

    test_simd_load_in_global(glb_data);
    test_simd_load_in_spm(glb_data);
    // 输出性能数据
    perf_print();
}

int main()
{
    // 设置使用0号SPA
    sdaaSetDevice(0);
    int h_glb_data[16] = { 0 };

    for (int i = 0; i < 16; i++){
        h_glb_data[i] = i;
    }

    int *glb_data = NULL;
    // 在Global内存中malloc一段空间
    sdaaMalloc((void **)&glb_data, 16 * sizeof(int));
    // 将数据从Host端拷贝到Device端
    sdaaMemcpy(glb_data, h_glb_data, 16 * sizeof(int), sdaaMemcpyHostToDevice);

    // kernel函数调用
    test_func<<<1>>>(glb_data);
    // 等待kernel函数执行完成
    sdaaDeviceSynchronize();

    // 释放申请的Global内存
    sdaaFree(glb_data);
    return 0;
}
```

#### 优化效果

| 接口名称 | 接口性能（执行周期，单位：拍）<br>（源地址和目的地址的数据均位于 Global 存储空间） | 接口性能（执行周期，单位：拍）<br>（源地址和目的地址的数据均位于 SPM 存储空间） |
|----------|------------------------------------------------------------------------------------|------------------------------------------------------------------------------|
| simd_load | 758 | 12 |

在本示例中，源地址和目的地址的数据均位于 SPM 存储空间，比源地址和目的地址的数据均位于 Global 存储空间，向量接口性能快约 63 倍。
